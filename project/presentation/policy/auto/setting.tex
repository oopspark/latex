% ===============================  SLIDE BORDER  ===============================
% 왼쪽 세로 바 + 자동 중앙 세로선
\newcommand{\SlideBorderVer}[1]{%
  \AddToShipoutPicture*{%
    % 왼쪽 세로 직사각형 바
    \begin{tikzpicture}[remember picture,overlay]
      \fill[slidebg]
        (current page.south west)
        rectangle ($(current page.north west) + (#1pt,0)$);
    \end{tikzpicture}

    % 중앙 세로 하이라이트 선 (바 폭에 따라 이동)
    \begin{tikzpicture}[remember picture,overlay]
      \draw[line width=1pt, white, line cap=round]
        ($(current page.center) + (#1pt - 215pt,  0.5\textheight)$)
        --
        ($(current page.center) + (#1pt - 215pt, -0.5\textheight)$);
    \end{tikzpicture}
  }%
}

\newcommand{\SlideBorderHor}[1]{%
  \AddToShipoutPicture*{%
    % 왼쪽 세로 직사각형 바
    \begin{tikzpicture}[remember picture,overlay]
      \fill[slidebg]
        ($(current page.north west) + (0,0)$)
        rectangle ($(current page.south east) + (0,#1pt)$);
    \end{tikzpicture}

    % 중앙 세로 하이라이트 선 (바 폭에 따라 이동)
    \begin{tikzpicture}[remember picture,overlay]
      \draw[line width=1pt, white, line cap=round]
        ($(current page.center) + (0.7\textheight, #1pt - 145pt)$)
        --
        ($(current page.center) + (-0.7\textheight, #1pt - 145pt)$);
    \end{tikzpicture}
  }%
}

\newcommand{\firstgap}{26}


\newcommand{\slide}[4]{%
\newpage
\SlideBorderHor{250}

\small {\hfill {\textcolor{white}{#1}}
\\[-\dimexpr \firstgap pt -12pt \relax]}

\LARGE {\textcolor{white}{\textbf{#3}} \hfill \raisebox{-2pt}{{\large \textcolor{white}{#2}}}
}
\vfill
\Large {#4}
\vfill
}
