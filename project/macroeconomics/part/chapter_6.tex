\chapter{Specific Factor Model (Ricardo--Viner Model)}

\section{Basic Setup}

\begin{itemize}
    \item Three factors: Labor $L$, Capital $K_x, K_y$
    \item Two goods: $X, Y$
\end{itemize}

\begin{itemize}
    \item Labor is mobile across sectors
    \item Capital is sector-specific
\end{itemize}

\[
L = L_x + L_y
\]

Production functions:
\[
X = X(L_x, K_x), \quad Y = Y(L_y, K_y)
\]

---

\section{Firm Optimization}

\[
\max \pi_x = p_x X(L_x, K_x) - w L_x - r_x K_x
\]

\[
\max \pi_y = p_y Y(L_y, K_y) - w L_y - r_y K_y
\]

FOC:

\[
p_x \frac{\partial X}{\partial L_x} = w, \quad
p_x \frac{\partial X}{\partial K_x} = r_x
\]

\[
p_y \frac{\partial Y}{\partial L_y} = w, \quad
p_y \frac{\partial Y}{\partial K_y} = r_y
\]

---

\section{Labor Market Equilibrium}

\[
VMP_{Lx} = VMP_{Ly} = w
\]

---

\section{Equilibrium Structure}

\begin{itemize}
    \item Labor allocates across sectors until wages equalize
    \item Returns to capital differ across sectors
\end{itemize}

---

\section{Comparative Statics: Price Change}

Suppose:
\[
\frac{p_x}{p_y} \uparrow
\]

Then:

\begin{itemize}
    \item Labor shifts to sector $X$
    \item $L_x \uparrow, \quad L_y \downarrow$
\end{itemize}

\[
VMP_{Lx} \uparrow, \quad VMP_{Ly} \downarrow
\]

Thus:
\[
w \uparrow
\]

---

\section{Factor Returns}

\begin{itemize}
    \item Return to capital in $X$:
    \[
    r_x \uparrow
    \]

    \item Return to capital in $Y$:
    \[
    r_y \downarrow
    \]
\end{itemize}

---

\section{Income Distribution}

\begin{itemize}
    \item Labor: ambiguous (depends on consumption bundle)
    \item Capital in export sector: gains
    \item Capital in import sector: loses
\end{itemize}

---

\section{Capital Change}

\subsection{$K_y \uparrow$ (Y-sector capital increase)}

\begin{itemize}
    \item $Y \uparrow$
    \item $X \downarrow$
    \item Labor shifts toward $Y$
\end{itemize}

\[
L_y \uparrow, \quad L_x \downarrow
\]

\[
w \uparrow
\]

\[
r_y \downarrow, \quad r_x \downarrow
\]

---

\subsection{$K_x \uparrow$ (X-sector capital increase)}

\begin{itemize}
    \item $X \uparrow$
    \item $Y \downarrow$
\end{itemize}

---

\section{Labor Endowment Increase}

\[
L \uparrow
\]

\begin{itemize}
    \item Both sectors expand
    \item Marginal product of labor decreases
\end{itemize}

\[
w \downarrow
\]

\[
r_x \uparrow, \quad r_y \uparrow
\]

---

\section{Summary Table}

\[
\begin{array}{c|ccc}
 & K_x \uparrow & K_y \uparrow & L \uparrow \\
\hline
X & + & - & + \\
Y & - & + & + \\
r_x & + & - & + \\
r_y & - & + & + \\
w & + & + & - \\
\end{array}
\]

---

\section{International Labor Movement}

Labor moves between countries:

\[
VMP_L^A = VMP_L^B
\]

\begin{itemize}
    \item Migration equalizes wages
    \item World output increases
\end{itemize}

---

\section{Welfare Effects}

\begin{itemize}
    \item Global welfare increases
    \item Redistribution occurs across factors
\end{itemize}

---

\section{Interpretation}

\begin{itemize}
    \item Specific factors determine distributional conflict
    \item Trade creates winners and losers
\end{itemize}
